028    package com.edugility.jaxb;
029    
030    import javax.xml.bind.annotation.XmlRootElement; // for javadoc only
031    
032    import javax.xml.bind.annotation.adapters.XmlAdapter;
033    import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter; // for javadoc only
034    
035    /**
036     * An {@link XmlAdapter} that helps JAXB handle an otherwise
037     * unbindable type when it is known that that unbindable type is
038     * implemented in a given application by one and only one bindable
039     * type.
040     *
041     * <p>For example, perhaps you know that in a given application all
042     * {@code Foobar} instances will in fact be {@code Whizbang}
043     * instances&mdash;a class, let's say, that is annotated with {@link
044     * XmlRootElement} (and other JAXB annotations) and implements {@code
045     * Foobar}.  But you do not want to write your API in terms of {@code
046     * Whizbang}s&mdash;you want to write it in terms of {@code Foobar}s.
047     * That is, you want JAXB to behave as though somehow suddenly it
048     * could handle {@code Foobar} as a type that it can express as an XML
049     * construct.  (Because {@code Foobar} is an interface, JAXB cannot
050     * natively handle it.)</p>
051     *
052     * <p>If though, it is truly known that in your application all
053     * instances of {@code Foobar} that JAXB will be asked to bind are
054     * actually {@code Whizbang}s&mdash;a concrete, JAXB-bindable type,
055     * then you could annotate your {@code Foobar} class with the {@link
056     * XmlJavaTypeAdapter} annotation and specify this class as the {@link
057     * XmlAdapter} implementation that would ensure that all {@code
058     * Foobar}s would be treated as {@code Whizbang}s internally by
059     * JAXB.</p>
060     *
061     * <p>This class is "universal" only in that it can adapt any two
062     * classes provided that one is a subtype of the other.</p>
063     *
064     * <p><strong>Note:</strong> The order of the type parameters is
065     * deliberately reversed from that of the {@link XmlAdapter}
066     * class.</p>
067     *
068     * @param <UnmappableType> the interface type (or other type) that
069     * JAXB needs to be able to handle but can't yet
070     *
071     * @param <MappableType> the implementation type that JAXB can handle
072     * that implements the {@code <B>} type
073     *
074     * @author <a href="http://about.me/lairdnelson"
075     * target="_parent">Laird Nelson</a>
076     *
077     * @see XmlAdapter
078     *
079     * @see XmlJavaTypeAdapter
080     *
081     * @see <a
082     * href="https://jaxb.java.net/guide/Mapping_interfaces.html">The
083     * Unofficial JAXB Guide's section on mapping interfaces</a>
084     *
085     * @see <a
086     * href="http://blog.bdoughan.com/2010/07/xmladapter-jaxbs-secret-weapon.html">Blaise
087     * Doughan's article on <code>XmlAdapter</code> usage</a>
088     */
089    public abstract class UniversalXmlAdapter<UnmappableType, MappableType extends UnmappableType> extends XmlAdapter<MappableType, UnmappableType> {
090      
091      /**
092       * Creates a new {@link UniversalXmlAdapter}.
093       */
094      public UniversalXmlAdapter() {
095        super();
096      }
097    
098      /**
099       * Converts the supplied {@code mappableObject}&mdash;a type that
100       * JAXB knows how to handle natively&mdash;into an instance of its
101       * supertype&mdash;which JAXB may not know how to handle natively.
102       *
103       * <p>This method is called when transforming an XML document into
104       * Java objects.  You can think of the {@code mappableObject}
105       * parameter as representing the XML value.</p>
106       *
107       * <p>This implementation simply returns the supplied {@code
108       * mappableObject}.  The type parameters supplied to this class at
109       * construction time ensure that this is always a valid thing to
110       * do.</p>
111       *
112       * @param mappableObject the object to unmarshal or deserialize; may
113       * be {@code null}
114       *
115       * @return the supplied {@code mappableObject}
116       */
117      @Override
118      public final UnmappableType unmarshal(final MappableType mappableObject) {
119        return mappableObject;
120      }
121    
122      /**
123       * Converts the supplied {@code unmappableObject} into an object
124       * that JAXB <em>can</em> handle natively by attempting to simply
125       * cast the supplied {@code unmappableObject} to the return type of
126       * this method.
127       *
128       * <p>This conversion should succeed provided that the type
129       * constraints of this class are respected.</p>
130       *
131       * <p>This method is called when transforming a Java object graph
132       * into an XML document.  You can think of the {@code
133       * unmappableObject} parameter as representing a Java object in the
134       * graph.</p>
135       *
136       * @param unmappableObject an object that for some reason or another
137       * JAXB is incapable of handling natively; may be {@code null}
138       *
139       * @return the result of casting the supplied {@code
140       * unmappableObject} to the return type of this method
141       *
142       * @exception ClassCastException if for some reason the cast could
143       * not take place
144       */
145      @Override
146      @SuppressWarnings("unchecked")
147      public final MappableType marshal(final UnmappableType unmappableObject) {
148        return (MappableType)unmappableObject;
149      }
150    }
