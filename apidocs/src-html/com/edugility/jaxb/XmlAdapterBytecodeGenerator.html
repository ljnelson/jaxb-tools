045    /**
046     * A generator of bytecode for {@link UniversalXmlAdapter} subclasses.
047     *
048     * @author <a href="http://about.me/lairdnelson"
049     * target="_parent">Laird Nelson</a>
050     *
051     * @see #generate(String, String, String)
052     *
053     * @see UniversalXmlAdapter
054     */
055    public class XmlAdapterBytecodeGenerator {
056    
057      /**
058       * An empty {@code byte} array suitable for edge cases encountered
059       * by the {@link #generate(String, String, String)} method.  This
060       * field is never {@code null}.
061       */
062      private static final byte[] EMPTY_BYTE_ARRAY = new byte[0];
063    
064      /**
065       * A {@linkplain Formatter format string} used to name a {@link
066       * UniversalXmlAdapter} subclass.  This field may be {@code null}.
067       *
068       * @see #getAdapterClassNameTemplate()
069       *
070       * @see #setAdapterClassNameTemplate(String)
071       *
072       * @see #getAdapterClassName(String, String, String)
073       */
074      private String adapterClassNameTemplate;
075    
076      /**
077       * Creates a new {@link XmlAdapterBytecodeGenerator} and {@linkplain
078       * #setAdapterClassNameTemplate(String) sets the default adapter
079       * class name template} to be {@code %s.%sTo%sAdapter}.
080       */
081      public XmlAdapterBytecodeGenerator() {
082        super();
083        this.setAdapterClassNameTemplate("%s.%sTo%sAdapter");
084      }
085    
086      /**
087       * Returns a {@linkplain Formatter format string} for use by the
088       * {@link #getAdapterClassName(String, String, String)} method.
089       *
090       * <p>This method may return {@code null}.</p>
091       *
092       * @return a {@link String} representing a {@linkplain Formatter
093       * format string} or {@code null}
094       *
095       * @see #setAdapterClassNameTemplate(String)
096       *
097       * @see #getAdapterClassName(String, String, String)
098       */
099      public String getAdapterClassNameTemplate() {
100        return this.adapterClassNameTemplate;
101      }
102    
103      /**
104       * Sets the {@linkplain Formatter format string} for use by the
105       * {@link #getAdapterClassNameTemplate()} method.
106       *
107       * @param adapterClassNameTemplate the {@linkplain Formatter format
108       * string}; must not be {@code null}
109       *
110       * @exception IllegalArgumentException if {@code
111       * adapterClassNameTemplate} is {@code null}
112       *
113       * @see #getAdapterClassNameTemplate()
114       *
115       * @see #getAdapterClassName(String, String, String)
116       */
117      public void setAdapterClassNameTemplate(final String adapterClassNameTemplate) {
118        if (adapterClassNameTemplate == null) {
119          throw new IllegalArgumentException("adapterClassNameTemplate", new NullPointerException("adapterClassNameTemplate"));
120        }
121        this.adapterClassNameTemplate = adapterClassNameTemplate;
122      }
123    
124      /**
125       * Returns the name for a {@link UniversalXmlAdapter} subclass, as
126       * formatted appropriately for a package named by the supplied
127       * {@code packageName} parameter, a {@link Class} to be adapted
128       * named by the supplied {@code interfaceName} parameter and an
129       * implementation {@link Class} named by the supplied {@code
130       * className} parameter.
131       *
132       * <p>This method never returns {@code null}.</p>
133       *
134       * @param packageName the name of a package; must not be {@code
135       * null}
136       *
137       * @param interfaceName a valid Java {@linkplain Class#getName()
138       * class name}; must not be {@code null}
139       *
140       * @param className a valid Java {@linkplain Class#getName() class
141       * name}; must not be {@code null}
142       *
143       * @return a non-{@code null} Java {@linkplain Class#getName() class
144       * name}
145       *
146       * @exception IllegalArgumentException if any of the parameters is
147       * {@code null}
148       *
149       * @exception IllegalStateException if {@link
150       * #getAdapterClassNameTemplate()} returns {@code null}
151       *
152       * @see #getAdapterClassNameTemplate()
153       *
154       * @see #setAdapterClassNameTemplate(String)
155       */
156      public final String getAdapterClassName(final String packageName, final String interfaceName, final String className) {
157        if (packageName == null) {
158          throw new IllegalArgumentException("packageName", new NullPointerException("packageName"));
159        }
160        if (interfaceName == null) {
161          throw new IllegalArgumentException("interfaceName", new NullPointerException("interfaceName"));
162        }
163        if (className == null) {
164          throw new IllegalArgumentException("className", new NullPointerException("className"));
165        }
166        final String template = this.getAdapterClassNameTemplate();
167        if (template == null) {
168          throw new IllegalStateException("The adapterClassNameTemplate property was null");
169        }
170        return String.format(template, packageName, this.getSimpleName(interfaceName), this.getSimpleName(className));
171      }
172    
173      /**
174       * Returns the last segment of a period-separated name, or the
175       * supplied {@code name} itself if it is {@code null} or not
176       * period-separated.
177       *
178       * <p>This method may return {@code null}.</p>
179       *
180       * @param name the name to parse; may be {@code null}
181       *
182       * @return the last segment of the supplied {@code name} if it is
183       * period-separated, or the supplied {@code name} itself if it is
184       * either {@code null} or not period-separated
185       */
186      private static final String getSimpleName(final String name) {
187        String returnValue = null;
188        if (name != null) {
189          final int lastDotIndex = name.lastIndexOf('.');
190          if (lastDotIndex < 0) {
191            returnValue = name;
192          } else {
193            assert name.length() > lastDotIndex + 1;
194            returnValue = name.substring(lastDotIndex + 1);
195          }
196        }
197        return returnValue;
198      }
199    
200      /**
201       * Calls the {@link #generate(String, String, String)} method,
202       * supplying it with the supplied {@code adapterClassName}, the
203       * {@linkplain Class#getName() name} of the supplied {@code
204       * interfaceClass} and the {@linkplain Class#getName() name} of the
205       * supplied {@code implementationClass}, and returns its results.
206       *
207       * <p>This method does not validate that the supplied {@code
208       * interfaceClass} parameter} {@linkplain Class#isInterface() is an
209       * interface}, only that the supplied {@code implementationClass}
210       * {@linkplain Class#isAssignableFrom(Class) implements or extends}
211       * it.</p>
212       *
213       * <p>This method never returns {@code null}.</p>
214       *
215       * @param <I> a type to be adapted
216       *
217       * @param <C> an implementation type that implements the {@code <I>}
218       * type
219       *
220       * @param adapterClassName the name of the {@link
221       * UniversalXmlAdapter} subclass; must not be {@code null}
222       *
223       * @param interfaceClass the interface being adapted; must not be
224       * {@code null}
225       *
226       * @param implementationClass the {@link Class} that is (a) not an
227       * interface and (b) guaranteed to be an implementation of the
228       * supplied {@code interfaceClass}; must not be {@code null}
229       *
230       * @return a non-{@code null} array of {@code byte}s representing
231       * the {@link UniversalXmlAdapter} subclass suitable for feeding to
232       * {@link ClassLoader#defineClass(String, byte[], int, int)} or for
233       * serialization to storage as a valid class file
234       *
235       * @exception BadBytecode if <a
236       * href="http://javassist.org/">Javassist</a> had problems with
237       * bytecode generation
238       *
239       * @exception CannotCompileException if <a
240       * href="http://javassist.org/">Javassist</a> had problems with
241       * bytecode generation
242       *
243       * @exception IOException if there was a problem reading or writing
244       *
245       * @exception NotFoundException if <a
246       * href="http://javassist.org/">Javassist</a> had problems with
247       * bytecode generation
248       *
249       * @see #generate(String, String, String)
250       *
251       * @see #getAdapterClassName(String, String, String)
252       */
253      public final <I, C extends I> byte[] generate(final String adapterClassName, final Class<I> interfaceClass, final Class<C> implementationClass) throws BadBytecode, CannotCompileException, IOException, NotFoundException {
254        if (adapterClassName == null) {
255          throw new IllegalArgumentException("adapterClassName", new NullPointerException("adapterClassName"));
256        }
257        if (interfaceClass == null) {
258          throw new IllegalArgumentException("interfaceClass", new NullPointerException("interfaceClass"));
259        }    
260        if (implementationClass == null) {
261          throw new IllegalArgumentException("implementationClass", new NullPointerException("implementationClass"));
262        }
263        if (!(interfaceClass.isAssignableFrom(implementationClass))) {
264          throw new IllegalArgumentException("!(interfaceClass.isAssignableFrom(implementationClass)): !(" + interfaceClass.getName() + ".isAssignableFrom(" + implementationClass.getName() + "))");
265        }
266        return this.generate(adapterClassName, interfaceClass.getName(), implementationClass.getName());
267      }
268    
269      /**
270       * Generates and returns the bytecode for a {@link
271       * UniversalXmlAdapter} subclass with the supplied {@code
272       * adapterClassName} as its {@linkplain Class#getName() name} that
273       * relies upon the fact that {@link Object}s that implement an
274       * interface with the supplied {@code interfaceClassName} will be
275       * instances of a {@link Class} with the supplied {@code
276       * implementationClassName}.
277       *
278       * <p>This method does not validate that the supplied {@code
279       * interfaceClassName} parameter actually identifies a loadable
280       * {@link Class} that {@linkplain Class#isInterface() is an
281       * interface} nor that the supplied {@code implementationClassName}
282       * actually identifies a loadable {@link Class} that {@linkplain
283       * Class#isInterface() is not an interface} and an implementation of
284       * the interface named by the supplied {@code interfaceClassName}.
285       * Consequently the resulting bytecode may not be functional in
286       * certain cases.</p>
287       *
288       * <p>This method never returns {@code null}.</p>
289       *
290       * @param adapterClassName the name of the {@link
291       * UniversalXmlAdapter} subclass; must not be {@code null}
292       *
293       * @param interfaceClassName the name of the interface being
294       * adapted; must not be {@code null}
295       *
296       * @param implementationClassName the name of a {@link Class} that
297       * is (a) not an interface and (b) guaranteed to be an
298       * implementation of the interface designated by the {@code
299       * interfaceClassName} parameter; must not be {@code null}
300       *
301       * @return a non-{@code null} array of {@code byte}s representing
302       * the {@link UniversalXmlAdapter} subclass suitable for feeding to
303       * {@link ClassLoader#defineClass(String, byte[], int, int)} or for
304       * serialization to storage as a valid class file
305       *
306       * @exception BadBytecode if <a
307       * href="http://javassist.org/">Javassist</a> had problems with
308       * bytecode generation
309       *
310       * @exception CannotCompileException if <a
311       * href="http://javassist.org/">Javassist</a> had problems with
312       * bytecode generation
313       *
314       * @exception IOException if there was a problem reading or writing
315       *
316       * @exception NotFoundException if <a
317       * href="http://javassist.org/">Javassist</a> had problems with
318       * bytecode generation
319       *
320       * @see #getAdapterClassName(String, String, String)
321       */
322      public byte[] generate(final String adapterClassName, final String interfaceClassName, final String implementationClassName) throws BadBytecode, CannotCompileException, IOException, NotFoundException {
323        if (adapterClassName == null) {
324          throw new IllegalArgumentException("adapterClassName", new NullPointerException("adapterClassName"));
325        }
326        if (interfaceClassName == null) {
327          throw new IllegalArgumentException("interfaceClassName", new NullPointerException("interfaceClassName"));
328        }
329        if (implementationClassName == null) {
330          throw new IllegalArgumentException("implementationClassName", new NullPointerException("implementationClassName"));
331        }
332        
333        ClassPool classPool = this.getClassPool(adapterClassName);
334        if (classPool == null) {
335          classPool = ClassPool.getDefault();
336        }
337        assert classPool != null;
338    
339        final CtClass universalXmlAdapterCtClass = classPool.get(UniversalXmlAdapter.class.getName());
340        assert universalXmlAdapterCtClass != null;
341    
342        final CtClass adapterCtClass = classPool.makeClass(adapterClassName, universalXmlAdapterCtClass);
343        assert adapterCtClass != null;
344        final ClassFile adapterClassFile = adapterCtClass.getClassFile();
345        assert adapterClassFile != null;
346    
347        assert adapterClassFile.getAttribute(SignatureAttribute.tag) == null; // we just created it after all
348    
349        final SignatureAttribute adapterClassSignatureAttribute = new SignatureAttribute(adapterClassFile.getConstPool(), String.format("L%s<L%s;L%s;>;", UniversalXmlAdapter.class.getName().replace('.', '/'), interfaceClassName.replace('.', '/'), implementationClassName.replace('.', '/')));
350        adapterClassFile.addAttribute(adapterClassSignatureAttribute);
351    
352        assert adapterClassFile.getAttribute(SignatureAttribute.tag) != null;
353        
354        byte[] returnValue = adapterCtClass.toBytecode();
355        if (returnValue == null) {
356          returnValue = EMPTY_BYTE_ARRAY;
357        }
358        return returnValue;
359    
360      }

362      /**
363       * Returns a <a href="http://javassist.org/">Javassist</a> {@link
364       * ClassPool} that is appropriate for the supplied class name.
365       *
366       * <p>The default implementation of this method ignores the {@code
367       * className} parameter and returns the return value of {@link
368       * ClassPool#getDefault()}.  For nearly all cases, this is the
369       * correct behavior and this method should not be overridden.</p>
370       *
371       * <p>If overrides of this method opt to return {@code null}, the
372       * return value of {@link ClassPool#getDefault()} will be used
373       * internally instead.</p>
374       *
375       * <p>This method is {@code protected}, not {@code private}, only
376       * because the actual semantics of the {@link
377       * ClassPool#getDefault()} are underspecified.  This allows
378       * consumers of this class to override this behavior if they are
379       * expecting {@link ClassPool}s to be produced in different
380       * ways.</p>
381       *
382       * @param className the {@link Class#getName() class name} for which
383       * the returned {@link ClassPool} might be appropriate; may be
384       * {@code null} and may safely be ignored; provided for contextual
385       * information only
386       *
387       * @return a {@link ClassPool} instance, or {@code null}, since the
388       * {@link ClassPool#getDefault()} does not document whether it is
389       * prohibited from returning {@code null}
390       *
391       * @see ClassPool
392       *
393       * @see ClassPool#getDefault()
394       */
395      protected ClassPool getClassPool(final String className) {
396        return ClassPool.getDefault();
397      }
