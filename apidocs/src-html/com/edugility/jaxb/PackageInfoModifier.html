079      public Modification modify(final String pkg) throws CannotCompileException, IOException, NotFoundException {
080        if (this.logger != null && this.logger.isLoggable(Level.FINER)) {
081          this.logger.entering(this.getClass().getName(), "modify", pkg);
082        }
083        if (pkg == null) {
084          throw new IllegalArgumentException("pkg", new NullPointerException("pkg"));
085        }
086    
087        final String packageInfoClassName = String.format("%s.package-info", pkg);
088    
089        ClassPool classPool = this.getClassPool(packageInfoClassName);
090        if (classPool == null) {
091          classPool = ClassPool.getDefault();
092        }
093        assert classPool != null;
094    
095        Modification.Kind kind = Modification.Kind.UNMODIFIED;
096    
097        CtClass packageInfoCtClass = classPool.getOrNull(packageInfoClassName);
098        if (packageInfoCtClass == null) {      
099          packageInfoCtClass = classPool.makeClass(packageInfoClassName);
100          kind = Modification.Kind.GENERATED;
101        }
102        assert packageInfoCtClass != null;
103        
104        final boolean modified = this.installXmlJavaTypeAdapters(packageInfoCtClass);
105        if (modified && !kind.equals(Modification.Kind.GENERATED)) {
106          kind = Modification.Kind.MODIFIED;
107        }
108        
109        final byte[] bytes = packageInfoCtClass.toBytecode();
110        assert bytes != null;
111        assert bytes.length > 0;
112    
113        final Modification returnValue = new Modification(packageInfoCtClass, kind, bytes);
114    
115        if (this.logger != null && this.logger.isLoggable(Level.FINER)) {
116          this.logger.exiting(this.getClass().getName(), "modify", returnValue);
117        }
118        return returnValue;
119      }

148      /**
149       * Installs an {@link XmlJavaTypeAdapters} annotation on the
150       * supplied {@link CtClass}, which must represent {@code
151       * package-info.class}.
152       *
153       * @param packageInfoCtClass the result of calling, e.g., {@link
154       * ClassPool#get(String) ClassPool.get(this.getPackage() +
155       * "package-info")}; must not be {@code null}
156       *
157       * @exception ClassNotFoundException 
158       */
159      private final boolean installXmlJavaTypeAdapters(final CtClass packageInfoCtClass) throws NotFoundException {
160        if (packageInfoCtClass == null) {
161          throw new IllegalArgumentException("packageInfoCtClass", new NullPointerException("packageInfoCtClass"));
162        }
163        final String pkg = packageInfoCtClass.getPackageName();
164        if (pkg == null) {
165          throw new IllegalArgumentException("packageInfoCtClass", new IllegalStateException("packageInfoCtClass.getPackage() == null"));
166        }
167        if (!String.format("%s.package-info", pkg).equals(packageInfoCtClass.getName())) {
168          throw new IllegalArgumentException("Wrong CtClass: " + packageInfoCtClass);
169        }
170    
171        boolean modified = false;
172    
173        final ClassFile packageInfoClassFile = packageInfoCtClass.getClassFile();
174        assert packageInfoClassFile != null;
175    
176        final ConstPool constantPool = packageInfoClassFile.getConstPool();
177        assert constantPool != null;
178    
179        AnnotationsAttribute annotationsAttribute = (AnnotationsAttribute)packageInfoClassFile.getAttribute(AnnotationsAttribute.visibleTag);
180        if (annotationsAttribute == null) {
181          annotationsAttribute = new AnnotationsAttribute(constantPool, AnnotationsAttribute.visibleTag);
182          packageInfoClassFile.addAttribute(annotationsAttribute);
183          modified = true;
184        }
185        assert annotationsAttribute != null;
186    
187        Annotation adaptersAnnotation = annotationsAttribute.getAnnotation(XmlJavaTypeAdapters.class.getName());
188        if (adaptersAnnotation == null) {
189          ClassPool classPool = this.getClassPool(XmlJavaTypeAdapters.class.getName());
190          if (classPool == null) {
191            classPool = ClassPool.getDefault();
192          }
193          assert classPool != null;
194          CtClass xmlJavaTypeAdaptersCtClass = classPool.getOrNull(XmlJavaTypeAdapters.class.getName());
195          assert xmlJavaTypeAdaptersCtClass != null;
196          adaptersAnnotation = new Annotation(constantPool, xmlJavaTypeAdaptersCtClass);
197          modified = true;
198        } else if (adaptersAnnotation.getMemberValue("value") == null) {
199          final ArrayMemberValue amv = new ArrayMemberValue(constantPool);
200          amv.setValue(new AnnotationMemberValue[0]);
201          adaptersAnnotation.addMemberValue("value", amv);
202          modified = true;
203        }
204        assert adaptersAnnotation != null;
205        assert adaptersAnnotation.getMemberValue("value") != null;
206    
207        modified = this.installXmlJavaTypeAdapters(adaptersAnnotation, constantPool) || modified;
208    
209        /*
210         * THIS COMMENT IS UNDER REVISION
211         *
212         * You would think this line would be required ONLY in the case
213         * where the annotation itself was not found.  But you actually
214         * have to add it to its containing AnnotationsAttribute in ALL
215         * cases.  This doesn't make any sense.  See
216         * http://stackoverflow.com/questions/8689156/why-does-javassist-insist-on-looking-for-a-default-annotation-value-when-one-is/8689214#8689214
217         * for details.
218         *
219         * Additionally, you must re-add the annotation as the last
220         * operation here in all cases.  Otherwise the changes made by the
221         * installXmlJavaTypeAdapters() method above are not actually made
222         * permanent.
223         */
224        if (modified) {
225          annotationsAttribute.addAnnotation(adaptersAnnotation);
226        }
227    
228        return modified;
229      }

231      /**
232       * Given a Javassist {@link Annotation} representing the {@link
233       * XmlJavaTypeAdapters} annotation, does whatever is necessary to
234       * make such an {@link Annotation} contain an array of {@link
235       * AnnotationMemberValue} objects, each of which represents an
236       * {@link XmlJavaTypeAdapter} annotation that has been specified in
237       * accordance with the rules represented by the return value of the
238       * {@link #getBindings()} method.
239       *
240       * @param adaptersAnnotation a non-{@code null} Javassist {@link
241       * Annotation} representation of the {@link XmlJavaTypeAdapters}
242       * annotation
243       *
244       * @param constantPool a {@link ConstPool} instance for use in
245       * constructing new {@link Annotation} instances; Javassist
246       * documentation does not specify whether {@code null} is a
247       * permitted value or not
248       *
249       * @exception NotFoundException in extremely rare situations when a
250       * JDK class cannot be located, usually due to {@link ClassLoader}
251       * problems
252       */
253      private final boolean installXmlJavaTypeAdapters(final Annotation adaptersAnnotation, final ConstPool constantPool) throws NotFoundException {
254        if (this.logger != null && this.logger.isLoggable(Level.FINER)) {
255          this.logger.entering(this.getClass().getName(), "installXmlJavaTypeAdapters", new Object[] { adaptersAnnotation, constantPool });
256        }
257        if (adaptersAnnotation == null) {
258          throw new IllegalArgumentException("adaptersAnnotation", new NullPointerException("adaptersAnnotation"));
259        }
260        if (!XmlJavaTypeAdapters.class.getName().equals(adaptersAnnotation.getTypeName())) {
261          throw new IllegalArgumentException("Wrong annotation: " + adaptersAnnotation.getTypeName());
262        }
263    
264        boolean modified = false;
265    
266        Map<String, String> bindings = this.getBindings();
267        if (bindings != null && !bindings.isEmpty()) {
268          final Set<Entry<String, String>> bindingEntries = bindings.entrySet();
269          if (bindingEntries != null && !bindingEntries.isEmpty()) {
270    
271            ClassPool classPool = this.getClassPool(XmlJavaTypeAdapter.class.getName());
272            if (classPool == null) {
273              classPool = ClassPool.getDefault();
274            }
275            assert classPool != null;
276    
277            ArrayMemberValue adaptersHolder = (ArrayMemberValue)adaptersAnnotation.getMemberValue("value");
278    
279            final Map<String, Annotation> existingAnnotationMap = getExistingXmlJavaTypeAdapters(adaptersHolder);        
280    
281            final List<AnnotationMemberValue> adapters = new ArrayList<AnnotationMemberValue>();
282    
283            final CtClass xmlJavaTypeAdapterCtClass = classPool.get(XmlJavaTypeAdapter.class.getName());
284            assert xmlJavaTypeAdapterCtClass != null;
285    
286            for (final Entry<String, String> entry : bindingEntries) {
287              if (entry != null) {            
288                final String adapterClassName = entry.getValue();
289                if (adapterClassName != null) {
290                  
291                  final String typeName = entry.getKey();
292                  if (typeName == null) {
293                    // An @XmlJavaTypeAdapter annotation is supposed to
294                    // have a type() attribute.  If it does not, there's
295                    // nothing we can do.
296                    continue;
297                  }
298    
299                  // Find or create a new @XmlJavaTypeAdapter with a type() of typeName.
300                  Annotation adapterAnnotation = null;
301                  if (existingAnnotationMap != null && !existingAnnotationMap.isEmpty()) {
302                    adapterAnnotation = existingAnnotationMap.get(typeName);
303                  }
304                  
305                  if (adapterAnnotation != null) {
306                    // Preexisting
307                    final ClassMemberValue v = (ClassMemberValue)adapterAnnotation.getMemberValue("value");
308                    assert v != null;
309                    final String existingClassName = v.getValue();
310                    if (adapterClassName.equals(existingClassName)) {
311                      // The annotation is already correctly specified
312                      continue;
313                    }
314                  }
315    
316                  if (adapterAnnotation == null) {
317                    adapterAnnotation = this.newXmlJavaTypeAdapter(constantPool, typeName);
318                    modified = true;
319                  }
320                  assert adapterAnnotation != null;
321                  assert XmlJavaTypeAdapter.class.getName().equals(adapterAnnotation.getTypeName());
322                  assert typeName.equals(((ClassMemberValue)adapterAnnotation.getMemberValue("type")).getValue());
323    
324                  modified = setXmlAdapter(adapterAnnotation, adapterClassName) || modified;
325                  assert adapterClassName.equals(((ClassMemberValue)adapterAnnotation.getMemberValue("value")).getValue());
326    
327                  // We have just found or created an @XmlJavaTypeAdapter;
328                  // add it to the list of such values.  This superset
329                  // list will become the new list of @XmlJavaTypeAdapters
330                  // on the package-info class.
331                  adapters.add(new AnnotationMemberValue(adapterAnnotation, constantPool));
332                }
333              }
334            }
335            
336            // Take the array of @XmlJavaTypeAdapter instances and set it
337            // as the value for @XmlJavaTypeAdapters#value().
338            final AnnotationMemberValue[] values = adapters.toArray(new AnnotationMemberValue[adapters.size()]);
339    
340            if (adaptersHolder == null) {
341              adaptersHolder = new ArrayMemberValue(constantPool);
342              adaptersAnnotation.addMemberValue("value", adaptersHolder);
343              modified = true;
344            }
345            assert values != null;
346            adaptersHolder.setValue(values);
347    
348          }
349        }
350        return modified;
351      }

353      private final Annotation newXmlJavaTypeAdapter(final ConstPool constantPool, final String typeName) throws NotFoundException {
354    
355        ClassPool classPool = this.getClassPool(XmlJavaTypeAdapter.class.getName());
356        if (classPool == null) {
357          classPool = ClassPool.getDefault();
358        }
359        assert classPool != null;
360    
361        final CtClass xmlJavaTypeAdapterCtClass = classPool.getOrNull(XmlJavaTypeAdapter.class.getName());
362        assert xmlJavaTypeAdapterCtClass != null;
363        
364        final Annotation adapterAnnotation = new Annotation(constantPool, xmlJavaTypeAdapterCtClass);
365    
366        // Retrieve the "holder" for the type() annotation attribute
367        // ("Foo.class" in the following sample:
368        //
369        //  @XmlJavaTypeAdapter(type = Foo.class, value = FooToFooImplAdapter.class)
370        //
371        final ClassMemberValue typeClassHolder = (ClassMemberValue)adapterAnnotation.getMemberValue("type");
372    
373        // (Because @XmlJavaTypeAdapter is a JDK-supplied class, and
374        // Javassist is just reading it, we are guaranteed that the type()
375        // annotation attribute will be modeled in Javassist.)
376        assert typeClassHolder != null;
377    
378        // Set the holder's value, thus installing the annotation's type()
379        // value.
380        typeClassHolder.setValue(typeName);
381    
382        return adapterAnnotation;
383      }

385      private static final boolean setXmlAdapter(final Annotation adapterAnnotation, final String adapterClassName) {
386        if (adapterClassName == null) {
387          throw new IllegalArgumentException("adapterClassName", new NullPointerException("adapterClassName"));
388        }
389        if (adapterAnnotation == null) {
390          throw new IllegalArgumentException("adapterAnnotation", new NullPointerException("adapterAnnotation"));
391        }
392        if (!XmlJavaTypeAdapter.class.getName().equals(adapterAnnotation.getTypeName())) {
393          throw new IllegalArgumentException("adapterAnnotation does not represent " + XmlJavaTypeAdapter.class.getName());
394        }
395    
396        // Retrieve the "holder" for the value() annotation
397        // attribute ("FooToFooImplAdapter.class" in the
398        // following sample:
399        // 
400        //   @XmlJavaTypeAdapter(type = Foo.class, value = FooToFooImplAdapter.class)
401        //
402        final ClassMemberValue adapterClassHolder = (ClassMemberValue)adapterAnnotation.getMemberValue("value");
403    
404        // (Because @XmlJavaTypeAdapter is a JDK-supplied class,
405        // and Javassist is just reading it, we are guaranteed
406        // that the type() annotation attribute will be modeled
407        // in Javassist.)
408        assert adapterClassHolder != null;
409    
410        final String old = adapterClassHolder.getValue();
411    
412        // Set the holder's value, thus installing the
413        // annotation's value() value.
414        adapterClassHolder.setValue(adapterClassName);
415    
416        if (old == null) {
417          return adapterClassName != null;
418        }
419        return !old.equals(adapterClassName);
420      }

422      /**
423       * A narrow-purpose helper method that accepts an {@link
424       * ArrayMemberValue} parameter whose value is (semantically) the
425       * return value of the {@link XmlJavaTypeAdapters#value()}
426       * annotation attribute, and returns a {@link Map} of {@link
427       * Annotation} objects representing {@link XmlJavaTypeAdapter}
428       * annotations indexed by the {@linkplain XmlJavaTypeAdapter#type()
429       * types} that each of them governs.
430       *
431       * <p>This method never returns {@code null}.</p>
432       *
433       * @param adaptersHolder the {@link ArrayMemberValue} that resulted
434       * from calling {@link Annotation#getMemberValue(String)
435       * Annotation#getMemberValue("value")} on an {@link Annotation} that
436       * is Javassist's representation of the {@link XmlJavaTypeAdapters}
437       * annotation.  This parameter may be {@code null}.
438       *
439       * @return a {@link Map} of {@link Annotation} objects representing
440       * {@link XmlJavaTypeAdapter} annotations indexed by the {@linkplain
441       * XmlJavaTypeAdapter#type() types} that each of them governs; never
442       * {@code null}
443       */
444      private static final Map<String, Annotation> getExistingXmlJavaTypeAdapters(final ArrayMemberValue adaptersHolder) {
445        // Build a Map indexing existing @XmlJavaTypeAdapter annotations
446        // by the types that they govern.
447        final Map<String, Annotation> xmlJavaTypeAdapters = new HashMap<String, Annotation>();
448        if (adaptersHolder != null) {
449          final MemberValue[] rawMemberValue = adaptersHolder.getValue();
450          if (rawMemberValue != null && rawMemberValue.length > 0) {
451            for (final MemberValue mv : rawMemberValue) {
452              if (mv instanceof AnnotationMemberValue) {
453                final Annotation xmlJavaTypeAdapter = ((AnnotationMemberValue)mv).getValue();
454                if (xmlJavaTypeAdapter != null && XmlJavaTypeAdapter.class.getName().equals(xmlJavaTypeAdapter.getTypeName())) {
455                  final ClassMemberValue typeHolder = (ClassMemberValue)xmlJavaTypeAdapter.getMemberValue("type");
456                  if (typeHolder != null) {
457                    final String interfaceTypeName = typeHolder.getValue();
458                    if (interfaceTypeName != null) {
459                      xmlJavaTypeAdapters.put(interfaceTypeName, xmlJavaTypeAdapter);
460                    }
461                  }
462                }
463              }
464            }
465          }
466        }
467        return xmlJavaTypeAdapters;
468      }

478      public static final class Modification implements Serializable {
479    
480        private static final long serialVersionUID = 1L;
481    
482        public static enum Kind {
483          UNMODIFIED, GENERATED, MODIFIED;
484        }
485    
486        private final CtClass packageInfoCtClass;
487    
488        private final byte[] bytes;
489    
490        private final Kind kind;
491    
492        private Modification(final CtClass packageInfoCtClass, final Kind kind, final byte[] bytes) {
493          super();
494          if (packageInfoCtClass == null) {
495            throw new IllegalArgumentException("packageInfoCtClass", new NullPointerException("packageInfoCtClass"));
496          }
497          if (!"package-info".equals(packageInfoCtClass.getSimpleName())) {
498            throw new IllegalArgumentException("packageInfoCtClass must be a package-info class: " + packageInfoCtClass);
499          }
500          this.packageInfoCtClass = packageInfoCtClass;
501          assert packageInfoCtClass.isFrozen();
502          if (kind == null) {
503            this.kind = Kind.MODIFIED;
504          } else {
505            this.kind = kind;
506          }
507          if (bytes == null) {
508            this.bytes = new byte[0];
509          } else {
510            this.bytes = bytes;
511          }
512        }
513    
514        public final String getPackageName() {
515          return this.packageInfoCtClass.getPackageName();
516        }
517    
518        public final Kind getKind() {
519          return this.kind;
520        }
521    
522        public final byte[] toByteArray() {
523          return this.bytes;
524        }

526      }
