028    package com.edugility.jaxb;
029    
030    import java.lang.reflect.ParameterizedType;
031    import java.lang.reflect.Type;
032    import java.lang.reflect.TypeVariable;
033    
034    import javax.xml.bind.annotation.adapters.XmlAdapter;
035    
036    import org.junit.After;
037    import org.junit.Before;
038    import org.junit.Test;
039    
040    import static org.junit.Assert.*;
041    
042    public class TestCaseXmlAdapterBytecodeGenerator {
043    
044      private XmlAdapterBytecodeGenerator generator;
045    
046      public TestCaseXmlAdapterBytecodeGenerator() {
047        super();
048      }
049    
050      @Before
051      public void setUp() throws Exception {
052        this.generator = new XmlAdapterBytecodeGenerator();
053      }
054    
055      @Test
056      public void testGenerate() throws Exception {
057    
058        // Make the relevant XML adapter class bits.
059        final byte[] newClass = this.generator.generate("com.edugility.jaxb.PersonToPersonImplementationAdapter", Person.class, PersonImplementation.class);
060        assertNotNull(newClass);
061        assertTrue(newClass.length > 0);
062    
063        // Turn those class bits into a real Java class.
064        final Class<?> adapterClass = new ClassDefiner().define("com.edugility.jaxb.PersonToPersonImplementationAdapter", newClass);
065        assertNotNull(adapterClass);
066        assertEquals("com.edugility.jaxb.PersonToPersonImplementationAdapter", adapterClass.getName());
067        assertTrue(UniversalXmlAdapter.class.isAssignableFrom(adapterClass));
068    
069        // Make sure that new class "is a" UniversalXmlAdapter.
070        assertSame(UniversalXmlAdapter.class, adapterClass.getSuperclass());
071    
072        // Make sure all the generics work right in the new class.
073        // UniversalXmlAdapter is an abstract class, and so when we
074        // subclass it we can get some of the parameterized type
075        // information at runtime.  Test it to make sure it's the right
076        // stuff.
077        final Type genericSuperclass = adapterClass.getGenericSuperclass();
078        assertNotNull(genericSuperclass);
079        assertTrue(genericSuperclass instanceof ParameterizedType);
080        final ParameterizedType adapterSuperType = (ParameterizedType)genericSuperclass;
081        assertSame(UniversalXmlAdapter.class, adapterSuperType.getRawType());
082        final Type[] actualTypeArguments = adapterSuperType.getActualTypeArguments();
083        assertNotNull(actualTypeArguments);
084        assertEquals(2, actualTypeArguments.length);
085        assertSame(Person.class, actualTypeArguments[0]);
086        assertSame(PersonImplementation.class, actualTypeArguments[1]);
087    
088        final UniversalXmlAdapter adapter = (UniversalXmlAdapter)adapterClass.newInstance();
089        assertNotNull(adapter);
090      }
091    
092      private static final class ClassDefiner extends ClassLoader {
093        
094        public final Class<?> define(final String className, final byte[] classBytes) throws Exception {
095          final Class<?> c = this.defineClass(className, classBytes, 0, classBytes.length);
096          assertNotNull(c);
097          this.resolveClass(c);
098          return c;
099        }
100    
101      }
102    
103    }
