103        validateTwoAnnotations(mod);
104      }
105    
106      private final void validateOneAnnotation(final PackageInfoModifier.Modification mod) throws Exception {
107        assertNotNull(mod);
108        final byte[] newClass = mod.toByteArray();
109        assertNotNull(newClass);
110        assertTrue(newClass.length > 0);
111    
112        final Class<?> c = new ClassDefiner().define(mod.getPackageName(), newClass);
113        assertNotNull(c);
114    
115        final Annotation[] annotations = c.getAnnotations();
116        assertNotNull(annotations);
117        assertTrue(annotations.length == 1);
118    
119        Annotation a = annotations[0];
120        assertNotNull(a);
121        assertTrue(a instanceof XmlJavaTypeAdapters);
122        final XmlJavaTypeAdapters adaptersAnnotation = (XmlJavaTypeAdapters)a;
123        final XmlJavaTypeAdapter[] adapters = adaptersAnnotation.value();
124        assertNotNull(adapters);
125        assertEquals(1, adapters.length);
126        final XmlJavaTypeAdapter adapter = adapters[0];
127        assertNotNull(adapter);
128        assertEquals(Person.class, adapter.type());
129        assertEquals(AnyTypeAdapter.class, adapter.value());
130      }
131    
132      private final void validateTwoAnnotations(final PackageInfoModifier.Modification mod) throws Exception {
133        assertNotNull(mod);
134        final byte[] newClass = mod.toByteArray();
135        assertNotNull(newClass);
136        assertTrue(newClass.length > 0);
137    
138        final Class<?> c = new ClassDefiner().define(mod.getPackageName(), newClass);
139        assertNotNull(c);
140        
141        final Annotation[] annotations = c.getAnnotations();
142        assertNotNull(annotations);
143        assertTrue(annotations.length == 2);
144    
145        Annotation a = annotations[0];
146        assertNotNull(a);
147        if (a instanceof Deprecated) {
148          a = annotations[1];
149          assertNotNull(a);
150        }
151        assertTrue(a instanceof XmlJavaTypeAdapters);
152        final XmlJavaTypeAdapters adaptersAnnotation = (XmlJavaTypeAdapters)a;
153        final XmlJavaTypeAdapter[] adapters = adaptersAnnotation.value();
154        assertNotNull(adapters);
155        assertEquals(1, adapters.length);
156        final XmlJavaTypeAdapter adapter = adapters[0];
157        assertNotNull(adapter);
158        assertEquals(Person.class, adapter.type());
159        assertEquals(AnyTypeAdapter.class, adapter.value());
160      }

191      private static final class ClassDefiner extends ClassLoader {
192    
193        public final Class<?> define(final String packageName, final byte[] classBytes) throws Exception {
194          final Class<?> c = this.defineClass(String.format("%s.package-info", packageName), classBytes, 0, classBytes.length);
195          assertNotNull(c);
